18bool BME280::init() {
19 if (!i2c_port) {
20 std::cerr << "Invalid I2C port.\n";
21 return false;
22 }
23
24 // Check device ID to confirm it's a BME280
25 uint8_t reg = 0xD0; // Chip ID register
26 uint8_t chip_id = 0;
27 int ret = i2c_write_blocking(i2c_port, device_addr, &reg, 1, true);
28 if (ret != 1) {
29 std::cerr << "Failed to write to BME280.\n";
30 return false;
31 }
32 ret = i2c_read_blocking(i2c_port, device_addr, &chip_id, 1, false);
33 if (ret != 1) {
34 std::cerr << "Failed to read chip ID from BME280.\n";
35 return false;
36 }
37 if (chip_id != 0x60) {
38 std::cerr << "Device is not a BME280.\n";
39 return false;
40 }
41
42 // Configure sensor
43 if (!configure_sensor()) {
44 std::cerr << "Failed to configure BME280 sensor.\n";
45 return false;
46 }
47
48 // Retrieve calibration parameters
49 if (!get_calibration_parameters()) {
50 std::cerr << "Failed to retrieve calibration parameters.\n";
51 return false;
52 }
53
54 initialized = true;
55 std::cout << "BME280 sensor initialized successfully.\n";
56 return true;

143bool BME280::get_calibration_parameters() {
