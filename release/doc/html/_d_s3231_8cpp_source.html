182uint8_t DS3231::bin_to_bcd(const uint8_t data) {
183 uint8_t ones_digit = (uint8_t)(data % 10);
184 uint8_t twos_digit = (uint8_t)(data - ones_digit) / 10;
185 return ((twos_digit << 4) + ones_digit);
186}
