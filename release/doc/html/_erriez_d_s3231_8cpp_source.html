74bool ErriezDS3231::clockEnable(bool enable)
75{
76 uint8_t reg;
77
78 // Read status register
79 reg = readRegister(DS3231_REG_CONTROL);
80
81 // Set or clear EOSC bit in control register
82 if (enable) {
83 reg &= ~(1 << DS3231_CTRL_EOSC);
84 } else {
85 reg |= (1 << DS3231_CTRL_EOSC);
86 }
87
88 // Write control register
89 if (!writeRegister(DS3231_REG_CONTROL, reg)) {
90 return false;
91 }
92
93 // Clear OSF bit in status register
94 reg = readRegister(DS3231_REG_STATUS);
95 reg &= ~(1 << DS3231_STAT_OSF);
96 return writeRegister(DS3231_REG_STATUS, reg);
97}

187bool ErriezDS3231::read(struct tm *dt)
188{
189 uint8_t buffer[7];
190
191 // Read clock date and time registers
192 if (!readBuffer(0x00, buffer, sizeof(buffer))) {
193 memset(dt, 0, sizeof(struct tm));
194 return false;
195 }
196
197 // Clear dt
198 memset(dt, 0, sizeof(struct tm));
199
200 // Convert BCD buffer to Decimal
201 dt->tm_sec = bcdToDec(buffer[0] & 0x7F);
202 dt->tm_min = bcdToDec(buffer[1] & 0x7F);
203 dt->tm_hour = bcdToDec(buffer[2] & 0x3F);
204 dt->tm_wday = bcdToDec(buffer[3] & 0x07);
205 dt->tm_mday = bcdToDec(buffer[4] & 0x3F);
206 dt->tm_mon = bcdToDec(buffer[5] & 0x1F);
207 dt->tm_year = bcdToDec(buffer[6]) + 100; // 2000-1900
208
209 // Month: 0..11
210 if (dt->tm_mon) {
211 dt->tm_mon--;
212 }
213
214 // Day of the week: 0=Sunday
215 if (dt->tm_wday) {
216 dt->tm_wday--;
217 }
218
219 // Check buffer for valid data
220 if ((dt->tm_sec > 59) || (dt->tm_min > 59) || (dt->tm_hour > 23) ||
221 (dt->tm_mday < 1) || (dt->tm_mday > 31) || (dt->tm_mon > 11) || (dt->tm_year > 199) ||
222 (dt->tm_wday > 6))
223 {
224 return false;
225 }
226
227 return true;
228}

243bool ErriezDS3231::write(const struct tm *dt)
244{
245 uint8_t buffer[7];
246
247 // Enable oscillator
248 if (!clockEnable(true)) {
249 return false;
250 }
251
252 // Encode date time from decimal to BCD
253 buffer[0] = decToBcd(dt->tm_sec) & 0x7F;
254 buffer[1] = decToBcd(dt->tm_min) & 0x7F;
255 buffer[2] = decToBcd(dt->tm_hour) & 0x3F;
256 buffer[3] = decToBcd(dt->tm_wday + 1) & 0x07;
257 buffer[4] = decToBcd(dt->tm_mday) & 0x3F;
258 buffer[5] = decToBcd(dt->tm_mon + 1) & 0x1F;
259 buffer[6] = decToBcd(dt->tm_year % 100);
260
261 // Write BCD encoded buffer to RTC registers
262 return writeBuffer(0x00, buffer, sizeof(buffer));
263}

306bool ErriezDS3231::getTime(uint8_t *hour, uint8_t *min, uint8_t *sec)
307{
308 uint8_t buffer[3];
309
310 // Read RTC time registers
311 if (!readBuffer(0x00, buffer, sizeof(buffer))) {
312 return false;
313 }
314
315 // Convert BCD buffer to Decimal
316 *sec = bcdToDec(buffer[0] & 0x7F);
317 *min = bcdToDec(buffer[1] & 0x7F);
318 *hour = bcdToDec(buffer[2] & 0x3F);
319
320 // Check buffer for valid data
321 if ((*sec > 59) || (*min > 59) || (*hour > 23)) {
322 // Invalid time
323 *sec = 0x00;
324 *min = 0x00;
325 *hour = 0x00;
326
327 return false;
328 }
329
330 return true;
331}

430bool ErriezDS3231::getTemperature(int8_t *temperature, uint8_t *fraction)
431{
432 uint8_t temp[2];
433
434 // Read temperature MSB and LSB registers
435 if (!readBuffer(DS3231_REG_TEMP_MSB, &temp, sizeof(temp))) {
436 return false;
437 }
438
439 // Set temperature argument
440 *temperature = temp[0];
441
442 // Calculate two's complement when negative
443 if (*temperature & 0x80) {
444 *temperature |= ~((1 << 8) - 1);
445 }
446
447 // Shift fraction bits 6 and 7 with 0.25 degree Celsius resolution
448 *fraction = (temp[1] >> 6) * 25;
449
450 return true;
451}

530bool ErriezDS3231::writeBuffer(uint8_t reg, void *buffer, uint8_t writeLen)
531{
532 // Start I2C transfer by writing the I2C address, register number and optional buffer
533 Wire.beginTransmission(DS3231_ADDR);
534 Wire.write(reg);
535 for (uint8_t i = 0; i < writeLen; i++) {
536 Wire.write(((uint8_t *)buffer)[i]);
537 }
538 if (Wire.endTransmission(true) != 0) {
539 return false;
540 }
541
542 return true;
543}

558bool ErriezDS3231::readBuffer(uint8_t reg, void *buffer, uint8_t readLen)
559{
560 // Start I2C transfer by writing the I2C address and register number
561 Wire.beginTransmission(DS3231_ADDR);
562 Wire.write(reg);
563 // Generate a repeated start, followed by a read buffer
564 if (Wire.endTransmission(false) != 0) {
565 return false;
566 }
567 Wire.requestFrom((uint8_t)DS3231_ADDR, readLen);
568 for (uint8_t i = 0; i < readLen; i++) {
569 ((uint8_t *)buffer)[i] = (uint8_t)Wire.read();
570 }
571
572 return true;
573}
