205int32_t INA3221::estimateOffsetVoltage(ina3221_ch_t channel, uint32_t busV) {
206 float bias_in = 10.0; // Input bias current at IN– in uA
207 float r_in = 0.670; // Input resistance at IN– in MOhm
208 uint32_t adc_step = 40; // smallest shunt ADC step in uV
209 float shunt_res = _shuntRes[channel]/1000.0; // convert to Ohm
210 float filter_res = _filterRes[channel];
211 int32_t offset = 0.0;
212 float reminder;
213
214 offset = (shunt_res + filter_res)*(busV/r_in + bias_in) - bias_in * filter_res;
215
216 // Round the offset to the closest shunt ADC value
217 reminder = offset % adc_step;
218 if (reminder < adc_step/2)
219 {
220 offset -= reminder;
221 } else {
222 offset += adc_step - reminder;
223 }
224
225 return offset;
226}
