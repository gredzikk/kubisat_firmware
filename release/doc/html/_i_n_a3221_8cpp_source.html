282int32_t INA3221::get_shunt_voltage(ina3221_ch_t channel) {
283 int32_t res;
284 ina3221_reg_t reg;
285 uint16_t val_raw = 0;
286
287 switch(channel){
288 case INA3221_CH1:
289 reg = INA3221_REG_CH1_SHUNTV;
290 break;
291 case INA3221_CH2:
292 reg = INA3221_REG_CH2_SHUNTV;
293 break;
294 case INA3221_CH3:
295 reg = INA3221_REG_CH3_SHUNTV;
296 break;
297 }
298
299 _read(reg, &val_raw);
300
301 res = (int16_t) (val_raw >> 3);
302 res *= SHUNT_VOLTAGE_LSB_UV;
303
304 return res;
305}

513void INA3221::_read(ina3221_reg_t reg, uint16_t *val) {
514 uint8_t reg_buf = reg;
515 uint8_t data[2];
516
517 int ret = i2c_write_blocking(MAIN_I2C_PORT, _i2c_addr, &reg_buf, 1, true);
518 if (ret != 1) {
519 std::cerr << "Failed to write register address to I2C device." << std::endl;
520 return;
521 }
522
523 ret = i2c_read_blocking(MAIN_I2C_PORT, _i2c_addr, data, 2, false);
524 if (ret != 2) {
525 std::cerr << "Failed to read data from I2C device." << std::endl;
526 return;
527 }
528
529 *val = (data[0] << 8) | data[1];
530}

539void INA3221::_write(ina3221_reg_t reg, uint16_t *val) {
540 uint8_t buf[3];
541 buf[0] = reg;
542 buf[1] = (*val >> 8) & 0xFF; // MSB
543 buf[2] = (*val) & 0xFF; // LSB
544
545 int ret = i2c_write_blocking(MAIN_I2C_PORT, _i2c_addr, buf, 3, false);
546 if (ret != 3) {
547 std::cerr << "Failed to write data to I2C device." << std::endl;
