65FileHandle fs_open_file(const char* filename, const char* mode) {
66 FileHandle handle = {-1, false};
67 std::string errorStr;
68
69 // Convert mode string to flags
70 int flags = 0;
71 if (strchr(mode, 'r')) flags |= O_RDONLY;
72 if (strchr(mode, 'w')) flags |= O_WRONLY | O_CREAT | O_TRUNC;
73 if (strchr(mode, 'a')) flags |= O_WRONLY | O_CREAT | O_APPEND;
74 if (strchr(mode, '+')) flags = O_RDWR | (flags & ~(O_RDONLY | O_WRONLY));
75
76 // Open the file
77 handle.fd = open(filename, flags, 0666);
78 if (handle.fd >= 0) {
79 handle.is_open = true;
80 } else {
81 errorStr = "Failed to open file " + std::string(filename) + ": " + strerror(errno);
82 uartPrint(errorStr);
83 }
84
85 return handle;
86}
