60FileHandle fs_open_file(const char* filename, const char* mode) {
61 FileHandle handle = {-1, false};
62
63 // Convert mode string to flags
64 int flags = 0;
65 if (strchr(mode, 'r')) flags |= O_RDONLY;
66 if (strchr(mode, 'w')) flags |= O_WRONLY | O_CREAT | O_TRUNC;
67 if (strchr(mode, 'a')) flags |= O_WRONLY | O_CREAT | O_APPEND;
68 if (strchr(mode, '+')) flags = O_RDWR | (flags & ~(O_RDONLY | O_WRONLY));
69
70 // Open the file
71 handle.fd = open(filename, flags, 0666);
72 if (handle.fd >= 0) {
73 handle.is_open = true;
74 } else {
75 printf("Failed to open file %s: %s\n", filename, strerror(errno));
76 }
77
78 return handle;
79}
